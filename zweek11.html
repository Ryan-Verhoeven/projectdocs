Code Voorbeelden

1. Basis Transition (Hover Effect)

<style>
    .button {
        background: blue;
        transition: all 0.3s ease;
    }
    
    .button:hover {
        background: darkblue;
        transform: translateY(-5px);
        box-shadow: 0 5px 15px rgba(0,0,0,0.3);
    }
</style>

2. Fade In Animatie

<style>
    @keyframes fadeIn {
        from {
            opacity: 0;
        }
        to {
            opacity: 1;
        }
    }
    
    .fade-in {
        animation: fadeIn 1s ease-in;
    }
</style>

3. Slide In van Links

<style>
    @keyframes slideIn {
        from {
            transform: translateX(-100%);
            opacity: 0;
        }
        to {
            transform: translateX(0);
            opacity: 1;
        }
    }
    
    .slide-in {
        animation: slideIn 0.5s ease-out;
    }
</style>

4. Bounce Effect

<style>
    @keyframes bounce {
        0%, 100% {
            transform: translateY(0);
        }
        25% {
            transform: translateY(-30px);
        }
        50% {
            transform: translateY(-15px);
        }
        75% {
            transform: translateY(-7px);
        }
    }
    
    .bounce {
        animation: bounce 0.6s ease;
    }
</style>

5. Rotate Animatie

<style>
    @keyframes rotate {
        from {
            transform: rotate(0deg);
        }
        to {
            transform: rotate(360deg);
        }
    }
    
    .rotate {
        animation: rotate 1s ease-in-out;
    }
</style>

6. Loading Spinner

<style>
    .spinner {
        width: 50px;
        height: 50px;
        border: 5px solid #e0e7ff;
        border-top: 5px solid #667eea;
        border-radius: 50%;
        animation: spin 1s linear infinite;
    }
    
    @keyframes spin {
        to {
            transform: rotate(360deg);
        }
    }
</style>

7. Pulse Effect (Infinite)

<style>
    @keyframes pulse {
        0%, 100% {
            transform: scale(1);
            opacity: 1;
        }
        50% {
            transform: scale(1.1);
            opacity: 0.8;
        }
    }
    
    .pulse {
        animation: pulse 1s infinite;
    }
</style>

8. JavaScript Animatie Toggle

<button onclick="animate()">Animeer!</button>
<div id="box"></div>

<script>
    function animate() {
        const box = document.getElementById('box');
        box.classList.add('fade-in');
        
        // Verwijder class na animatie
        setTimeout(() => {
            box.classList.remove('fade-in');
        }, 1000);
    }
</script>
